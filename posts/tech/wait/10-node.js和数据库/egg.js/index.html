config.sequelize = {
    datasources: [
      {
        delegate: 'model', // 加载所有的模型到 app.model and ctx.model
        baseDir: 'model', // 要加载的模型目录`app/model/*.js`
        dialect: 'mysql',
        host: '10.52.18.8',
        port: 3306,
        username: 'root', // 用户名
        password: 'root', // 用户密码
        database: 'crt_crland_screen', // 数据库名
        dateStrings: true, // 解决时间格式
        pool: {
          max: 20,
          min: 0,
          idle: 10000,
        },
      },
      {
        delegate: 'appModel', // 加载所有的模型到 app.logModel and ctx.logModel
        baseDir: 'model_app', // 要加载的模型目录 `app/model_log/*.js`
        // local
        dialect: 'mysql',
        host: '10.52.18.8',
        port: 3306,
        username: 'root', // 用户名
        password: 'root', // 用户密码
        database: 'hotel_report', // 数据库名
        dateStrings: true, // 解决时间格式
        pool: {
          max: 20,
          min: 0,
          idle: 10000,
        },
      },
    ],
  };

注意事项：

很多人担心 controller 、service 、 model 需要怎么编写

controller \ service 正常编写

model 需要注意 后续添加的数据库指定了baseDir（新的model 目录）

配置的delegate 属性 ==> app.appModel / ctx.appModel （连接表 / 服务时记得把model 换成配置的 delegate ）
