模糊查询关键字高亮

解决方案
1 min · 2019/01/01
<template>
	<view>
    	<view class="search">
			<image class="search-icon" src="../../../static/zfdygl-search@2x.png" mode=""></image>
			<input type="text" placeholder="请输入行业名称/代码" v-model="searchIdCard" maxlength="18" 	   					@input="handInputSearch" @confirm="searchIDcardConfirm"confirm-type="搜索" />
		</view>
		<view class="search-content">
			<view class="li one-line-overflow" v-for="(sItem,sIndex) in searchList" :key="sIndex">
				<text v-html="sItem.industry_code"></text>-
				<text v-html="sItem.industry_name"></text>
			</view>
		</view>
    </view>    
</template>


methods:{
	brightKeyword(val) {
		let keyword = this.searchIdCard.toString();
		if (val.indexOf(keyword) !== -1) {
			return val.replace(keyword, `<span style="color:#3995FF;font-weight:bold;">${keyword}</span>`)
		} else {
					return val
		}
	},
	handInputSearch(e){
		clearTimeout(this.searchTime);
		this.searchTime = setTimeout(() => {
			this.searchForm.industry_code = e.detail.value;
			this.searchForm.parent_indu = "";
			this.getDataist(this.searchForm);
		}, 500)
    },
    getDataist(){
        Service.apiIndustryList(fd).then(res => {
			if (res.RetCode == 100) {
				this.searchList = [];

				res.data.industry.forEach(i => {
					i.industry_name = this.brightKeyword(i.industry_name)
					i.industry_code = this.brightKeyword(i.industry_code)
				})
				this.searchList = res.data.industry
		}).catch(error => {
				console.log(error)
		})
    }
}
