    <div>
        <ChildrenVue :msg='msg' />
    </div>

<script setup >
import ChildrenVue from '../components/Children.vue';

  const msg = ref('哈喽哈喽')
}
</script

<template>
    <div>
        <h1> {{msg}}</h1>
    </div>
</template>
<script setup>
export default {
    props: ['msg'],
    const msg = props.msg
}
</script>

//父组件
<template>
    <ChildrenVue @myClick='onMyClick' />
</template>
<script setup>
import ChildrenVue from '../components/Children.vue';
const onMyClick = (msg) => {
    console.log(msg);
}
</script>

//子组件
<template>
    <button @click="handleClick">按钮</button>
</template>
<script setup>
import { defineEmits } from 'vue'
const emit = defineEmits(['myClick'])
const handleClick = () => {
    emit('myClick', '我是子组件给父组件的')
}
</script>

  path: '/detail',
  query: {
    id: 12345
  }
})

const route = useRoute()
const id = computed(() => {
  return route.query.id
})

//父组件(传值组件)
<script setup>
   import { provide } from "vue"
   provide("name", "lvxiaobu")
</script>

//子组件(接收组件)
<script setup>
   import { inject } from "vue"
   const name = inject("name")
   console.log(name) // lvxiaobu
</script>
